# 配置含义介绍

```json
{
  "$schema": "https://json.schemastore.org/prettierrc",
  // 去掉分号
  "semi": false,
  // 使用单引号
  "singleQuote": true,
  // 超过100换行
  "printWidth": 100,
  // 使用2个空格缩进
  "tabWidth": 2,
  // 不使用tab缩进
  "useTabs": false,
  // 多行对象/数组等末尾保留逗号，diff 更干净
  "trailingComma": "all",
  // 对象/导入括号内留空格
  "bracketSpacing": true,
  // 多行标签的 > 另起一行
  "bracketSameLine": false,
  // 箭头函数参数总带括号
  "arrowParens": "always",
  // 模板属性多时一行一个
  "singleAttributePerLine": true,
  // 减少模板空白导致的抖动
  "htmlWhitespaceSensitivity": "ignore",
  // SFC 的 <script>/<style> 内缩进
  "vueIndentScriptAndStyle": true,
  // 嵌入块能格式化就格式化
  "embeddedLanguageFormatting": "auto",
  // Markdown 等不强制自动换行
  "proseWrap": "preserve",
  // 统一换行符为 LF
  "endOfLine": "lf"
}
```
